C (构造)

题意

构造一个 \(n \times m\) 的矩阵，使得每一个 \(2 \times 2\) 的子矩阵满足： \[ \begin{matrix} A_{11} \oplus A_{12} \oplus A_{21} \oplus A_{22}=A_{33}\oplus A_{34} \oplus A_{43} \oplus A_{44} \\A_{13} \oplus A_{14} \oplus A_{23} \oplus A_{24}=A_{31}\oplus A_{32} \oplus A_{41} \oplus A_{42} \end{matrix} \]

题解

通过观察样例，可以猜想每一个子矩阵的异或和是一个常数。（待证明）
因此，我们可以随机化矩阵的第一行和第一列，并设定一个常数 \(C\) 作为每个子矩阵的异或和。对于一个子矩阵，若已知三个数的值 \(A_{1}, A_{2}, A_{3}\)，则第四个值 \(A_{4}=C \oplus A_{1} \oplus A_{2} \oplus A_{3}\)。

python 代码

import random
T = int(input())
mod = (1 << 63) - 1
f = [[0 for _ in range(200)] for _ in range(200)]
for i in range(200):
    f[i][0] = random.randint(0, 100000000000000012)
    f[0][i] = random.randint(0, 100000000000000012)
for i in range(1, 200):
    for j in range(1, 200):
        f[i][j] = mod ^ f[i - 1][j] ^ f[i - 1][j - 1] ^ f[i][j - 1]
for _ in range(T):
    n, m = [int(_) for _ in input().split()]
    print(n*m)
    for i in range(n):
        for j in range(m):
            print('%d ' % f[i][j], end='')
        print()

D (贪心、multiset)

题意

给出两个数组 \(a, b\)，操作 \(n\) 次，对于第 \(i\) 次，每次只能从 \(a_{i}、b_{i}\) 中选出一个，记从 \(a\) 随选出的数列为 \(a_{choose}\)，从 \(b\) 中选出的数列为 \(b_{choose}\)，目标是最小化 $| (a_{choose}) - (b_{choose}) | $。

题解

将 \(a, b\) 一起按照 \(a\) 从大到小排序，从前往后遍历 \(a\)，对于 \(a_{i}\)，将其作为从 \(a\) 中选出的最大值。
由于 \(\forall j \in [1,i-1], a_{j} < a_{i}\)，因此，\(i\) 前面的 \(a\) 一定不选，\(i\) 前面的 \(b\) 一定要选；\(i\) 后面的 \(b\) 可选可不选。
记 \(mxb = \max(b_{j}, j \in [1,i-1])\) ，则只需要在 \(i\) 后面的 \(b\) 中找到和 \(a_{i}\) 最相近的且 \(> mxb\) 的数 \(b_{k}\) 即可用 \(\left | a_{i} - b_{k} \right |\) 更新答案；如果没有找到，则用 \(\left | a_{i} - mxb \right |\) 更新答案。
可用 \(muliset\) 维护，\(lower_bound\) 用于寻找最相近的数。
时间复杂度 \(O(nlogn)\)

c++ 代码

#include <iostream>
#include <set>
#include <algorithm>
#define _for( i, L, R ) for ( int i = L; i <= R; ++i )

using namespace std;

const int N = 5e5 + 5;

class Node {
public:
	int a, b;
	bool operator<( const Node & o ) const
	{
		return(a > o.a);
	}
} e[N];
int n;

void solve()
{
	scanf( "%d", &n );
	multiset<int> s;
	_for( i, 1, n ) scanf( "%d%d", &e[i].a, &e[i].b ), s.insert( e[i].b );
	sort( e + 1, e + n + 1 );
	int res = 1e9, mx = -1e9;
	_for( i, 1, n )
	{
		s.erase( s.find( e[i].b ) );
		auto it = s.lower_bound( e[i].a );
		if ( it != s.end() and * it > mx )
			res = min( res, abs( e[i].a - *it ) );
		if ( it != s.begin() and * prev( it ) > mx )
			res = min( res, abs( e[i].a - *prev( it ) ) );
		res	= min( res, abs( e[i].a - mx ) );
		mx	= max( e[i].b, mx );
	}
	printf( "%d\n", res );
}


int main()
{
	int T;
	scanf( "%d", &T );
	while ( T-- )
		solve();
	return(0);
}

E (树状数组优化dp)

题意

给出 \(n\) 个数组，每个数组有 \(k_{i}\) 个数，找到一直数组拼接方式，使得拼接后的数组 \(b\) 中满足 \(\forall j \in [1, i-1], b_{i} > b_{j}\) （条件1）的数最多。

题解

考虑dp。记 \[ f[i]：前 i 个数组拼接而成的最大答案 \]

  • 由于一个数组中可能对答案产生贡献的只有在本数组中满足条件1的数，因此在读入时即可使数组单调递增
  • 将 \(n\) 个数组按照最后一个元素从小到大排序，这样在更新 \(f[i]\) 时，前面的状态一定已经被更新完了
  • 对于 \(f[i]\)，其中的某一个数 \(a_{i, j}\)，贡献来自两部分
    • 由于 \(\forall p \in [j+1, k_{i}], a_{i, p} > a_{i, j}\)，因此贡献为 \(k_{i}-j\)
    • 对于最大值小于 \(a_{i, j}\) 的数组 \(a_{s}\)，其贡献为 \(\max(f[s])\)

因此，状态转移方程为
\[ f[i] = \max(\max(f[s]) + k_{i} - j), s = \{s| \max(a_{s}) < a_{i, j}\} \] 其中，\(\max(f[s])\) 可用树状数组维护。
时间复杂度 \(O(nklogn)\)

c++ 代码

#pragma GCC optimize(1)
#pragma GCC optimize(2)
#pragma GCC optimize(3,"Ofast","inline")
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
#define _for( i, L, R ) for ( int i = L; i <= R; ++i )

using namespace std;

inline int lowbit( int x )
{
	return(x & - x);
}


const int	N = 2e5 + 5;
int		n, tr[N], f[N], mxv;
class Node {
public:
	vector<int> v;
	bool operator<( const Node & o ) const
	{
		return(v.back() < o.v.back() );
	}
} a[N];

void modify( int x, int y )
{
	for ( int i = x; i <= mxv; i += lowbit( i ) )
		tr[i] = max( tr[i], y );
}


int query( int x )
{
	int res = 0;
	for ( int i = x; i; i -= lowbit( i ) )
		res = max( res, tr[i] );
	return(res);
}


void clear( int x )
{
	for ( int i = x; i <= mxv; i += lowbit( i ) )
		tr[i] = 0;
}


void solve()
{
	scanf( "%d", &n );
	mxv = 0;
	_for( i, 1, n )
	{
		int m;
		scanf( "%d", &m );
		while ( m-- )
		{
			int x;
			scanf( "%d", &x );
			if ( a[i].v.empty() or a[i].v.back() < x )
				a[i].v.push_back( x ), mxv = max( mxv, x );
		}
	}
	sort( a + 1, a + 1 + n );
	int res = 0;
	_for( i, 1, n )
	{
		int k = a[i].v.size(), cnt = 0;
		for ( int j = k - 1; j >= 0; --j )
		{
			cnt++;
			f[i] = max( f[i], cnt + query( a[i].v[j] - 1 ) );
		}
		modify( a[i].v.back(), f[i] );
		res = max( res, f[i] );
	}
	_for( i, 1, n ) clear( a[i].v.back() ), a[i].v.clear(), f[i] = 0;
	printf( "%d\n", res );
}


int main()
{
	int T = 1;
	scanf( "%d", &T );
	while ( T-- )
		solve();
	return(0);
}
